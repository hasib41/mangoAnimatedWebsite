JS

// it always blows out
function surge() {
  tl.to(root, {
    '--heat': 1 })      // over-drive
    .add(brownOut())   // flicker
    .call(arcFlash)    // it severs
    .call(() => {
      coil.hide();
      embers.show();    // red→black
      soot.bloom();
    });
}
cord.on('pull', surge);
