Cute Lamp

CSS

.light {
  opacity: 0;
  transition: opacity .12s;
  filter: drop-shadow(
    0 0 40px #ffe6a3);
}
.lamp.on .light {
  opacity: 1 }

JS

// pull the cord to toggle
const lamp = $('.lamp')
const cord = $('.cord')
cord.onpointerdown = drag
function drag(e) {
  if (pull(e) > DETENT)
    lamp.classList.toggle('on')
}
